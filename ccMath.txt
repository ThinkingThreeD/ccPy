--------------------      LERP  ---------------------------------------
-------------------------------------------------------------------------
Houdini SDK
SYSlerp(fpreal32 v1, fpreal32 v2, fpreal32 t)
{
return v1 + (v2 - v1)*t;
}

-------------------------------------------------------------------------
--------------------  CLAMP  ---------------------------------------
-------------------------------------------------------------------------
Maya SDK
inline double clamp(double a, double l, double h)
{
    return ((a) < (l) ? (l) : (a) > (h) ? (h) : (a));
}

Houdini SDK
Clamp(Type x, Type min, Type max)
{
  assert(min<max);
  return x > min ? x < max ? x : max : min;
}
